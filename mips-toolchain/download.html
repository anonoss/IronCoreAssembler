Quick Start Guide

Get up and running in minutes

1 Linux Installation 

# Debian/Ubuntu
sudo dpkg -i mipsx_2.0.0_amd64.deb

# Red Hat/Fedora
sudo rpm -i mipsx-2.0.0-1.x86_64.rpm

# Verify installation
mipsx --help

2 Windows Build 

# Extract source archive
# Navigate to build directory
cd mipsx-2.0.0\build
build_windows.bat

# The executable will be in dist\mips.exe

3 macOS Build 

# Extract and build
cd mipsx-2.0.0/build
./build_macos.sh

# Install
sudo cp dist/mipsx /usr/local/bin/

4 Install VS Code Extension 

code --install-extension mips-x-2.0.0.vsix

5 Usage 

# Run MIPS assembly file
mipsx run program.asm

# Build to hex
mipsx build program.asm -o output.hex

# In VS Code: Press Ctrl+Shift+R to run
